### 2.2 MCP Server实现（伪代码）

```javascript
// 在淘宝客户端主进程中
const express = require('express');
const { BrowserWindow } = require('electron');

class TaobaoMCPServer {
  constructor() {
    this.port = 3654;
    this.app = express();
    this.mainWindow = null; // 淘宝主窗口引用
    this.tools = this.registerTools();
  }

  registerTools() {
    return {
      navigate: this.handleNavigate.bind(this),
      read_page_content: this.handleReadPage.bind(this),
      click_element: this.handleClickElement.bind(this),
      input_text: this.handleInputText.bind(this),
      // ... 更多工具
    };
  }

  async handleNavigate({ url }) {
    // 通过IPC发送导航指令到渲染进程
    return await this.mainWindow.webContents.executeJavaScript(`
      window.location.href = '${url}';
      new Promise(resolve => {
        window.addEventListener('load', () => resolve({ success: true }));
      });
    `);
  }

  async handleReadPage() {
    // 通过DevTools Protocol获取页面内容
    const cdp = await this.mainWindow.webContents.debugger;
    const { result } = await cdp.sendCommand('Runtime.evaluate', {
      expression: `document.body.innerText`,
      returnByValue: true
    });
    return { content: result.value };
  }

  async handleClickElement({ element_id }) {
    // 在渲染进程中执行点击
    return await this.mainWindow.webContents.executeJavaScript(`
      const element = document.querySelector('[data-mcp-id="${element_id}"]');
      if (element) {
        element.click();
        return { success: true };
      }
      return { error: 'Element not found' };
    `);
  }

  start() {
    this.app.post('/mcp', this.handleMCPRequest.bind(this));
    this.app.listen(this.port);
    console.log(`MCP Server running on port ${this.port}`);
  }

  async handleMCPRequest(req, res) {
    const { method, params } = req.body;
    
    // MCP协议路由
    switch (method) {
      case 'tools/list':
        return res.json({ tools: Object.keys(this.tools) });
      
      case 'tools/call':
        const { name, arguments: args } = params;
        const result = await this.tools[name](args);
        return res.json({ result });
      
      default:
        return res.status(400).json({ error: 'Unknown method' });
    }
  }
}

// 启动MCP服务器
const mcpServer = new TaobaoMCPServer();
mcpServer.start();
```

### 2.3 mcporter中间件实现

```javascript
// mcporter的核心功能
class MCPorter {
  constructor(config) {
    this.mcpServers = config.mcpServers;
  }

  async callTool(serverName, toolName, args) {
    const server = this.mcpServers[serverName];
    
    if (server.type === 'streamableHttp') {
      return await this.streamableHttpCall(server.url, toolName, args);
    }
  }

  async streamableHttpCall(url, toolName, args) {
    const response = await fetch(url, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        method: 'tools/call',
        params: {
          name: toolName,
          arguments: args
        }
      })
    });

    return await response.json();
  }
}
```

## 三、MCP协议实现

### 3.1 MCP协议规范

MCP (Model Context Protocol) 是一个标准化的AI-工具交互协议：

```json
{
  "protocol": "MCP/1.0",
  "capabilities": {
    "tools": true,
    "resources": false,
    "prompts": false
  }
}
```

### 3.2 taobao-native支持的MCP消息类型

| 方法 | 说明 | 示例 |
|------|------|------|
| `initialize` | 建立连接，协商能力 | 客户端启动时调用 |
| `tools/list` | 获取可用工具列表 | 返回所有可用工具 |
| `tools/call` | 调用指定工具 | 执行具体操作 |
| `notifications/*` | 服务器推送通知 | 页面状态变化等 |

```javascript
// taobao-native使用客户端的Cookie
class AuthManager {
  async getCurrentUser() {
    const cookies = await mainWindow.webContents.session.cookies.get({
      domain: '.taobao.com'
    });
    
    // 从Cookie中提取用户信息
    const userCookie = cookies.find(c => c.name === '_tb_token_');
    return {
      token: userCookie?.value,
      isLoggedIn: !!userCookie
    };
  }

  // MCP工具调用时自动带上登录态
  async callWithAuth(toolFn, args) {
    const user = await this.getCurrentUser();
    if (!user.isLoggedIn) {
      throw new Error('User not logged in');
    }
    return await toolFn(args);
  }
}
```

```javascript
// 淘宝客户端内部的工具注册表
class ToolRegistry {
  constructor() {
    this.tools = new Map();
  }

  register(name, schema, handler) {
    this.tools.set(name, {
      name,
      description: schema.description,
      inputSchema: schema.inputSchema,
      handler
    });
  }

  async call(name, args) {
    const tool = this.tools.get(name);
    if (!tool) {
      throw new Error(`Tool ${name} not found`);
    }

    // 参数验证
    this.validateArgs(args, tool.inputSchema);

    // 执行工具
    return await tool.handler(args);
  }

  validateArgs(args, schema) {
    // JSON Schema验证
    // ...
  }

  list() {
    return Array.from(this.tools.values()).map(tool => ({
      name: tool.name,
      description: tool.description,
      inputSchema: tool.inputSchema
    }));
  }
}
```

### 5.3 核心工具实现原理

#### navigate（页面导航）

```javascript
async function navigate({ url }) {
  // 1. URL验证（安全检查）
  if (!url.startsWith('https://') || !url.includes('taobao.com')) {
    throw new Error('Only taobao.com URLs are allowed');
  }

  // 2. 执行导航
  await mainWindow.loadURL(url);

  // 3. 等待页面稳定
  await new Promise(resolve => {
    mainWindow.webContents.once('did-finish-load', resolve);
  });

  // 4. 注入页面标记（用于后续元素定位）
  await mainWindow.webContents.executeJavaScript(`
    // 给所有可交互元素添加唯一ID
    document.querySelectorAll('button, a, input').forEach((el, idx) => {
      el.setAttribute('data-mcp-id', 'mcp-el-' + idx);
    });
  `);

  return { success: true, url };
}
```

#### scan_page_elements（扫描元素）

```javascript
async function scan_page_elements({ selector }) {
  // 扫描所有可交互元素
  const elements = await mainWindow.webContents.executeJavaScript(`
    Array.from(document.querySelectorAll('${selector || 'button, a, input'}')).map(el => ({
      id: el.getAttribute('data-mcp-id'),
      tag: el.tagName.toLowerCase(),
      type: el.type,
      text: el.textContent?.trim().substring(0, 50),
      classes: Array.from(el.classList),
      
      // 位置信息
      rect: {
        x: el.getBoundingClientRect().x,
        y: el.getBoundingClientRect().y,
        width: el.getBoundingClientRect().width,
        height: el.getBoundingClientRect().height
      },
      
      // 可见性
      visible: el.offsetParent !== null,
      
      // 可点击性
      clickable: !el.disabled && el.offsetParent !== null
    }))
  `);

  return { elements };
}
```

#### click_element（点击元素）

```javascript
async function click_element({ element_id }) {
  // 1. 滚动到元素可见区域
  await mainWindow.webContents.executeJavaScript(`
    const el = document.querySelector('[data-mcp-id="${element_id}"]');
    el.scrollIntoView({ behavior: 'smooth', block: 'center' });
  `);

  // 等待滚动完成
  await new Promise(resolve => setTimeout(resolve, 500));

  // 2. 获取元素坐标
  const rect = await mainWindow.webContents.executeJavaScript(`
    document.querySelector('[data-mcp-id="${element_id}"]').getBoundingClientRect()
  `);

  // 3. 模拟真实鼠标点击
  await mainWindow.webContents.sendInputEvent({
    type: 'mouseMove',
    x: Math.floor(rect.x + rect.width / 2),
    y: Math.floor(rect.y + rect.height / 2)
  });

  await mainWindow.webContents.sendInputEvent({
    type: 'mouseDown',
    x: Math.floor(rect.x + rect.width / 2),
    y: Math.floor(rect.y + rect.height / 2),
    button: 'left',
    clickCount: 1
  });

  await mainWindow.webContents.sendInputEvent({
    type: 'mouseUp',
    x: Math.floor(rect.x + rect.width / 2),
    y: Math.floor(rect.y + rect.height / 2),
    button: 'left',
    clickCount: 1
  });

  // 4. 或者直接调用click()（更简单但不够真实）
  // await mainWindow.webContents.executeJavaScript(`
  //   document.querySelector('[data-mcp-id="${element_id}"]').click();
  // `);

  return { success: true };
}
```

#### input_text（输入文本）

```javascript
async function input_text({ element_id, text }) {
  // 1. 聚焦输入框
  await mainWindow.webContents.executeJavaScript(`
    const el = document.querySelector('[data-mcp-id="${element_id}"]');
    el.focus();
  `);

  // 2. 清空现有内容
  await mainWindow.webContents.executeJavaScript(`
    const el = document.querySelector('[data-mcp-id="${element_id}"]');
    el.value = '';
  `);

  // 3. 逐字符输入（模拟真实打字）
  for (const char of text) {
    await mainWindow.webContents.sendInputEvent({
      type: 'char',
      keyCode: char
    });
    await new Promise(resolve => setTimeout(resolve, 50)); // 打字延迟
  }

  // 4. 触发input事件（确保React等框架检测到变化）
  await mainWindow.webContents.executeJavaScript(`
    const el = document.querySelector('[data-mcp-id="${element_id}"]');
    el.dispatchEvent(new Event('input', { bubbles: true }));
    el.dispatchEvent(new Event('change', { bubbles: true }));
  `);

  return { success: true };
}
```

```javascript
class SecurityManager {
  // 1. URL白名单
  isAllowedURL(url) {
    const allowedDomains = [
      'taobao.com',
      's.taobao.com',
      'detail.tmall.com',
      'cart.taobao.com'
    ];
    return allowedDomains.some(domain => url.includes(domain));
  }

  // 2. 敏感操作限制
  isSensitiveOperation(toolName) {
    const sensitiveOps = ['submit_order']; // 下单需要二次确认
    return sensitiveOps.includes(toolName);
  }

  // 3. 频率限制
  async checkRateLimit(toolName) {
    const key = `ratelimit:${toolName}`;
    const count = await this.redis.incr(key);
    await this.redis.expire(key, 60); // 1分钟窗口
    
    if (count > 30) { // 每分钟最多30次
      throw new Error('Rate limit exceeded');
    }
  }

  // 4. 支付确认（特殊处理）
  async handlePayment(orderData) {
    // 不允许AI直接支付，必须用户手动确认
    return {
      status: 'pending',
      message: 'Please complete payment manually',
      paymentUrl: '...'
    };
  }
```

### 8.3 并发控制

```javascript
class ConcurrencyManager {
  constructor(maxConcurrent = 3) {
    this.maxConcurrent = maxConcurrent;
    this.running = 0;
    this.queue = [];
  }

  async execute(fn) {
    if (this.running >= this.maxConcurrent) {
      await new Promise(resolve => this.queue.push(resolve));
    }
    
    this.running++;
    try {
      return await fn();
    } finally {
      this.running--;
      const next = this.queue.shift();
      if (next) next();
    }
  }
}

// 使用
const concurrency = new ConcurrencyManager(3);
await concurrency.execute(() => navigate({ url: '...' }));
```

## 九、与浏览器自动化的本质区别

### 9.1 技术架构对比

| 维度 | taobao-native | Playwright/Puppeteer |
|------|---------------|----------------------|
| **实现层级** | 应用层（Electron进程内） | 协议层（CDP外部连接） |
| **权限** | 完整客户端权限 | 浏览器沙箱内权限 |
| **登录态** | 直接使用客户端登录 | 需要管理Cookie |
| **反检测** | 无需反检测 | 需要绕过各种检测 |
| **API稳定性** | 官方维护，长期稳定 | 依赖页面结构 |
| **代码注入** | 主进程直接注入 | 通过CDP注入 |

### 9.2 工作原理对比

**taobao-native (官方方案)**:
```javascript
// 在Electron主进程中
mainWindow.webContents.executeJavaScript(`
  // 拥有完整权限，不受CSP限制
  document.querySelector('.buy-btn').click();
`);
// ✅ 可以访问所有DOM
// ✅ 可以调用任何客户端API
// ✅ 与客户端登录态完全一致
```

**Playwright (外部自动化)**:
```javascript
// 通过CDP协议
await page.evaluate(`
  // 受到CSP和沙箱限制
  document.querySelector('.buy-btn').click();
`);
// ⚠️ 可能被页面CSP阻止
// ⚠️ 需要绕过反爬虫检测
// ⚠️ 需要手动管理Cookie
```

### 9.3 检测机制对比

**浏览器自动化会被检测的特征**:
```javascript

navigator.webdriver === true // 自动化标记

// 5. 浏览器特征
// User-Agent、Canvas指纹、WebGL指纹等
```

**taobao-native不会被检测**:
```javascript

// - navigator.webdriver === undefined
// - 没有CDP痕迹
// - 鼠标事件可以完美模拟

```
